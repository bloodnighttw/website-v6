---
title: "Navigation Hook"
description: "Programmatically navigate between routes using the useNavigate hook."
---

The `useNavigate` hook provides programmatic navigation capabilities for client-side routing.

## Basic Usage

```tsx
import useNavigate from "rpress/navigate";

export default function LoginForm() {
  const navigate = useNavigate();

  const handleSubmit = async (formData: FormData) => {
    try {
      await login(formData);
      navigate("/dashboard");
    } catch (error) {
      console.error("Login failed:", error);
    }
  };

  return (
    <form action={handleSubmit}>
      <input name="email" type="email" required />
      <input name="password" type="password" required />
      <button type="submit">Login</button>
    </form>
  );
}
```

## Navigation After Actions

Navigate after form submissions or API calls:

```tsx
import useNavigate from "rpress/navigate";

export default function CreatePost() {
  const navigate = useNavigate();

  const handleCreatePost = async (title: string, content: string) => {
    try {
      const post = await createPost({ title, content });
      navigate(`/posts/${post.id}`);
    } catch (error) {
      console.error("Failed to create post:", error);
    }
  };

  return (
    <div>
      <h1>Create New Post</h1>
      <PostForm onSubmit={handleCreatePost} />
    </div>
  );
}
```

## Conditional Navigation

Navigate based on conditions:

```tsx
import useNavigate from "rpress/navigate";

export default function ProtectedPage() {
  const navigate = useNavigate();
  const { user, loading } = useAuth();

  useEffect(() => {
    if (!loading && !user) {
      navigate("/login");
    }
  }, [user, loading, navigate]);

  if (loading) return <div>Loading...</div>;
  if (!user) return null;

  return <div>Protected content</div>;
}
```

## Navigation with State

Navigate and manage application state:

```tsx
import useNavigate from "rpress/navigate";

export default function ShoppingCart() {
  const navigate = useNavigate();
  const { cart, clearCart } = useCart();

  const handleCheckout = () => {
    if (cart.items.length === 0) {
      alert("Cart is empty");
      return;
    }

    navigate("/checkout");
  };

  const handleContinueShopping = () => {
    navigate("/products");
  };

  return (
    <div>
      <h1>Shopping Cart</h1>
      {cart.items.map((item) => (
        <CartItem key={item.id} item={item} />
      ))}
      <button onClick={handleCheckout}>Checkout</button>
      <button onClick={handleContinueShopping}>Continue Shopping</button>
    </div>
  );
}
```

## Event Handlers

Use in event handlers for interactive navigation:

```tsx
import useNavigate from "rpress/navigate";

export default function ProductCard({ product }) {
  const navigate = useNavigate();

  return (
    <div className="product-card">
      <img
        src={product.image}
        alt={product.name}
        onClick={() => navigate(`/products/${product.id}`)}
      />
      <h3>{product.name}</h3>
      <p>${product.price}</p>
      <button onClick={() => navigate(`/products/${product.id}`)}>
        View Details
      </button>
    </div>
  );
}
```

## Navigation with Search Params

Navigate to routes with query parameters:

```tsx
import useNavigate from "rpress/navigate";

export default function SearchFilters() {
  const navigate = useNavigate();

  const handleFilterChange = (category: string, price: string) => {
    const params = new URLSearchParams();
    if (category) params.set("category", category);
    if (price) params.set("price", price);

    navigate(`/products?${params.toString()}`);
  };

  return (
    <div>
      <select onChange={(e) => handleFilterChange(e.target.value, "")}>
        <option value="">All Categories</option>
        <option value="electronics">Electronics</option>
        <option value="clothing">Clothing</option>
      </select>
    </div>
  );
}
```

## Comparison with Link

Use `useNavigate` for programmatic navigation and `Link` for user-initiated navigation:

```tsx
import Link from "rpress/link";
import useNavigate from "rpress/navigate";

export default function Navigation() {
  const navigate = useNavigate();

  // Use Link for user clicks
  const userNavigation = (
    <nav>
      <Link to="/about">About</Link>
      <Link to="/contact">Contact</Link>
    </nav>
  );

  // Use useNavigate for programmatic navigation
  const handleTimeout = () => {
    setTimeout(() => {
      navigate("/timeout");
    }, 5000);
  };

  return (
    <div>
      {userNavigation}
      <button onClick={handleTimeout}>Auto-redirect in 5s</button>
    </div>
  );
}
```
